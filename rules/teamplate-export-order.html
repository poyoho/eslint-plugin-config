此规则的正确代码示例：

<script>
export default {
  methods: {
    clickHello () {},
    clickWorld () {},

    enterInputName () {},
    enterInputEmail () {},
  },
  setup () {
    const varA = reactive()

    function clickHello() {}
    function enterInputName() {}

    const varB = reactive()

    function clickWorld() {}

    const varC = reactive()

    function enterInputEmail() {}

    return {
      clickHello,
      clickWorld,
      enterInputName,
      enterInputEmail,
      varA,
      varB,
      varC,
    }
  }
}
</script>
